package org.telran.lecture_01_intro.practice;

public class Max3 {
    public static void main(String[] args) {
        System.out.println(max3(22, 5, -4));
        System.out.println(max3(2, 55, 14));
        System.out.println(max3(12, 5, 44));
    }

    public static int max3(int a, int b, int c) {
        if (a <= b) {
            a = b; //b больше или равно a, тогда присваиваем b значение a
        }
        if (a > c) return a; //Если a больше c, то a - наибольшее число

        return c;
    }
}
